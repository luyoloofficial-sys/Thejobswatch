function setupSearch() { let input = document.getElementById("searchBox"); input.addEventListener("keyup", function () { let filter = input.value.toLowerCase(); let jobs = document.getElementsByClassName("job"); for (let i = 0; i < jobs.length; i++) { let text = jobs[i].innerText.toLowerCase(); if (text.includes(filter)) { jobs[i].style.display = ""; } else { jobs[i].style.display = "none"; } } }); }
